Also we can calculate median by splice (it modifies source array, so you should copy it, if want save)

    let array = [...[1, 23, 14, 432, 34]]
    array.sort((a, b) => a - b)
    let splice = array.splice(Math.ceil(array.length/2) - 1, 2 - (array.length % 2))
    let median = (splice[0] + (splice[1] || 0)) / splice.length
